\begin{lstlisting}[language=Python,caption={資料品質驗證腳本}]
import pandas as pd
from datetime import datetime, timedelta

def validate_price_data(df: pd.DataFrame) -> dict:
    """
    驗證價格數據品質

    Args:
        df: 包含 date, open, high, low, close, volume 的 DataFrame

    Returns:
        dict: 驗證結果摘要
    """
    results = {
        'coverage': 0,
        'continuity': True,
        'completeness': True,
        'duplicates': 0,
        'price_anomalies': 0,
        'issues': []
    }

    # 1. 覆蓋率檢查
    if 'date' in df.columns:
        date_range = (df['date'].max() - df['date'].min()).days
        actual_days = df['date'].nunique()
        # 假設工作日約佔 5/7
        expected_days = date_range * 5 / 7
        results['coverage'] = actual_days / expected_days if expected_days > 0 else 0

    # 2. 時間連續性 (檢查是否有超過 5 天的缺口)
    if 'date' in df.columns:
        df_sorted = df.sort_values('date')
        gaps = df_sorted['date'].diff()
        max_gap = gaps.max().days if gaps.max() else 0
        if max_gap > 5:
            results['continuity'] = False
            results['issues'].append(f'最大缺口: {max_gap} 天')

    # 3. 欄位完整性
    required_cols = ['open', 'high', 'low', 'close', 'volume']
    for col in required_cols:
        if col in df.columns:
            null_count = df[col].isna().sum()
            if null_count > 0:
                results['completeness'] = False
                results['issues'].append(f'{col} 有 {null_count} 個 NULL 值')

    # 4. 重複檢查
    if 'date' in df.columns:
        results['duplicates'] = df.duplicated(subset=['date']).sum()

    # 5. 價格異常 (負值或極端值)
    if 'close' in df.columns:
        anomalies = ((df['close'] <= 0) | (df['close'] > 10000)).sum()
        results['price_anomalies'] = anomalies

    # OHLC 邏輯檢查
    if all(c in df.columns for c in ['open', 'high', 'low', 'close']):
        ohlc_issues = (
            (df['high'] < df['low']) |
            (df['high'] < df['open']) |
            (df['high'] < df['close']) |
            (df['low'] > df['open']) |
            (df['low'] > df['close'])
        ).sum()
        if ohlc_issues > 0:
            results['issues'].append(f'OHLC 邏輯錯誤: {ohlc_issues} 筆')

    return results
